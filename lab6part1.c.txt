void main() {

    FILE *file;
    float numbers[12];
    int i = 0;
	char filename[30];
	scanf("%s", filename);
    // Open the input file in read mode
	    file = fopen(filename, "r");
    if (file == NULL) {
        printf("Error: Could not open file.\n");
        return 1;
    }

    // Read numbers from the file and store them in the array
    while (i < 12 && fscanf(file, "%f", &numbers[i]) == 1) {
        i++;
    }

    // Close the file
    fclose(file);
}
